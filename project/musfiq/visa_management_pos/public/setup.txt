<?php
// ডাটাবেস কানেকশন যুক্ত করা হলো
require_once '../app/config/database.php';

$message = '';

if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $username = trim($_POST['username']);
    $password = trim($_POST['password']);
    $role = 'superadmin';

    // পাসওয়ার্ড হ্যাশ করা (সিকিউরিটির জন্য)
    $hashed_password = password_hash($password, PASSWORD_DEFAULT);

    try {
        // ডাটাবেসে ইউজার আছে কি না চেক করা
        $checkStmt = $conn->prepare("SELECT id FROM users WHERE username = :username");
        $checkStmt->execute(['username' => $username]);

        if ($checkStmt->rowCount() > 0) {
            $message = "<div class='bg-red-100 text-red-700 p-3 rounded mb-4'>এই ইউজারনেমটি আগে থেকেই আছে!</div>";
        } else {
            // নতুন এডমিন ইনসার্ট করা
            $stmt = $conn->prepare("INSERT INTO users (username, password, role) VALUES (:username, :password, :role)");
            $stmt->execute([
                'username' => $username,
                'password' => $hashed_password,
                'role' => $role
            ]);
            $message = "<div class='bg-green-100 text-green-700 p-3 rounded mb-4 font-bold'>
                            Admin created successfully! <br> 
                            <span class='text-red-600'>সতর্কতা: সিকিউরিটির জন্য এখনই এই (setup_admin.php) ফাইলটি ডিলিট করে দিন!</span>
                        </div>";
        }
    } catch (PDOException $e) {
        $message = "<div class='bg-red-100 text-red-700 p-3 rounded mb-4'>Error: " . $e->getMessage() . "</div>";
    }
}
?>

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Create Admin</title>
    <script src="https://cdn.tailwindcss.com"></script>
</head>

<body class="bg-gray-200 flex items-center justify-center h-screen">

    <div class="bg-white p-8 rounded-lg shadow-lg w-full max-w-md border-t-4 border-red-500">
        <h2 class="text-2xl font-bold text-center text-gray-800 mb-2">Setup First Admin</h2>
        <p class="text-center text-sm text-red-500 mb-6 font-semibold">Delete this file after use!</p>

        <?php echo $message; ?>

        <form method="POST" action="">
            <div class="mb-4">
                <label class="block text-gray-700 text-sm font-bold mb-2">Admin Username</label>
                <input type="text" name="username" required
                    class="border rounded w-full py-2 px-3 text-gray-700 focus:outline-none focus:ring-2 focus:ring-blue-500"
                    placeholder="Enter username (e.g. admin)">
            </div>

            <div class="mb-6">
                <label class="block text-gray-700 text-sm font-bold mb-2">Admin Password</label>
                <input type="password" name="password" required
                    class="border rounded w-full py-2 px-3 text-gray-700 focus:outline-none focus:ring-2 focus:ring-blue-500"
                    placeholder="Enter strict password">
            </div>

            <button type="submit"
                class="bg-blue-600 hover:bg-blue-700 text-white font-bold py-2 px-4 rounded w-full">
                Create Admin User
            </button>

            <div class="mt-4 text-center">
                <a href="index.php" class="text-blue-500 hover:underline text-sm">Go to Login / Dashboard</a>
            </div>
        </form>
    </div>

</body>

</html>
